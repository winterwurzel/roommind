7def _is_fahrenheit(hass: HomeAssistant, entity_id: str | None = None) -> bool: 

8 """Check if the value is in Fahrenheit. 

9 

10 When *entity_id* is provided the entity's own ``unit_of_measurement`` 

11 attribute is used. This is race-condition-safe during HA unit-system 

12 changes because entity state + attributes are always updated atomically. 

13 Falls back to the global HA config when no entity is given. 

14 """ 

15 if entity_id: 

16 state = hass.states.get(entity_id) 

17 if state: 

18 uom = state.attributes.get("unit_of_measurement") 

19 if uom: 

20 return bool(uom == "°F") 

21 return bool(hass.config.units.temperature_unit == UnitOfTemperature.FAHRENHEIT)

24def ha_temp_to_celsius( 

25 hass: HomeAssistant, 

26 value: float, 

27 *, 

28 entity_id: str | None = None, 

29) -> float: 

30 """Convert temperature from HA unit system to Celsius. 

31 

32 Pass *entity_id* when reading from a sensor entity to avoid race 

33 conditions during unit-system changes. 

34 """ 

35 if _is_fahrenheit(hass, entity_id): 

36 return (value - 32) * 5 / 9 

37 return value

40def celsius_to_ha_temp(hass: HomeAssistant, value: float) -> float: 

41 """Convert temperature from Celsius to HA unit system.""" 

42 if hass.config.units.temperature_unit == UnitOfTemperature.FAHRENHEIT:
