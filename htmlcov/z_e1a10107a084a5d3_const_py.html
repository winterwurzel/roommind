195def is_override_active(room: dict) -> bool: 

196 """Return True when a manual override is currently active.""" 

197 override_temp = room.get("override_temp") 

198 if override_temp is None: 

199 return False 

200 override_until = room.get("override_until") 

201 return override_until is None or time.time() < override_until 

202 

203 

204def is_override_suppressed(room: dict, settings: dict, presence_away: bool) -> bool: 

205 """Return True when an active override should be ignored due to presence.""" 

206 if not presence_away: 

207 return False 

208 if room.get("ignore_presence", False): 

209 return False 

210 return bool(settings.get("presence_clears_override", False))

213def build_override_live(room: dict, suppressed: bool = False) -> dict: 

214 """Build override fields for live data from a room config dict. 

215 

216 When *suppressed* is True the override is held in the store but currently 

217 has no effect (e.g. presence-away with presence_clears_override enabled). 

218 Live data still reports the underlying intent so the UI can render a 

219 "paused" indicator. 

220 """ 

221 active = is_override_active(room) 

222 override_temp = room.get("override_temp") 

223 override_until = room.get("override_until") 

224 return { 

225 "override_active": active, 

226 "override_type": room.get("override_type") if active else None, 

227 "override_temp": override_temp if active else None, 

228 "override_until": override_until if active else None, 

229 "override_suppressed": active and suppressed, 

230 }
